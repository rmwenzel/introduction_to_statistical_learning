```python
from sklearn.metrics import confusion_matrix

                  name='Actual')
y_pred = pd.Series(['Yes' if prob[1] > 0.2 else 'No' for prob in boost_clf.predict_proba(X_test)],
                        name='Predicted')
boost_tree_conf = pd.crosstab(y_act, y_pred, margins=True)
```
